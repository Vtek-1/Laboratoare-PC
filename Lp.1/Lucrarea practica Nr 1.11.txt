#include<stdio.h>
#include<math.h>
int main()
{
float x1,x2,px,a,b,c;
float x,F;
int n;

printf("Imput x1: ");
scanf("%f", &x1);
printf("Imput x2: ");
scanf("%f", &x2 );
printf("Imput px: ");
scanf("%f", &px);
printf("Imput a: ");
scanf("%f", &a);
printf("Imput b: ");
scanf("%f", &b);
printf("Imput c: ");
scanf("%f", &c);
x = x1;
n = 0;
if (x>x2)
{
    printf("\n x1>x2. Condition false");
}
else
{
while (x < x2)
{
printf("\n\t Results: \n");
n = n + 1;
if((x+b<0 && c==0))
{
        F = (sin(a)+b*x)/fabs(pow(x,2)-c);
}
    else if ((x+b)>0 && c!=0)
    {
            F = (log10(a*x))/(pow(c,2)+b);
    }
        else
        {
                F = (4*sqrt(x)+8*cos(x))/(pow(a,2)+b);
        }
printf("%i:x=%.3f \t F=%.3f \n", n,x,F);
x = x + px;
}
}
    return 0;
}
