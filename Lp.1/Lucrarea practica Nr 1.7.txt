#include<stdio.h>
#include<math.h>
int main()
{
float x1,x2,px,a,b,c;
float x,F;
int n;

printf("Imput x1: ");
scanf("%f", &x1);
printf("Imput x2: ");
scanf("%f", &x2 );
printf("Imput px: ");
scanf("%f", &px);
printf("Imput a: ");
scanf("%f", &a);
printf("Imput b: ");
scanf("%f", &b);
printf("Imput c: ");
scanf("%f", &c);
x = x1;
n = 0;
if (x>x2)
{
    printf("\n x1>x2. Condition false");
}
else
{
while (x < x2)
{
printf("\n\t Results: \n");
n = n + 1;
if((x-2)>0 && a == 0)
{
        F = (a+log10(x)/sin(c)-(b*b));
}
    else if ((x-2)<0 && a != 0)
    {
            F = (x-sin(x+1))/b;
    }
        else
        {
                F = (a*x+c)/(cos(2*x));
        }
printf("%i:x=%.3f \t F=%.3f \n", n,x,F);

}
}
    return 0;
}
